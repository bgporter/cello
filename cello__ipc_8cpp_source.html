44IpcClient::IpcClient (Object& objectToWatch, UpdateType updateType, const juce::String& hostName, int portNum,
45 const juce::String& pipeName, int msTimeout, Object* state)
46: juce::InterprocessConnection { true, CelloMagicIpcNumber }
47, juce::ValueTreeSynchroniser { objectToWatch }
48, UpdateQueue { objectToWatch, nullptr }
49, clientProperties { objectToWatch.getType ().toString (), state }
50, update { updateType }

58 jassert ((update & UpdateType::send) || (update & UpdateType::receive));
59 // ...and if we are sending a full update, we also need to be sending (in general)
60 jassert (!(update & UpdateType::fullUpdateOnConnect) ||
61 ((update & UpdateType::fullUpdateOnConnect) && (update & UpdateType::send)));
62}

83bool IpcClient::connect (ConnectOptions options)
84{
85 if (host.isNotEmpty ())
86 {
87 // the options have no meaning for a socket connection, just try to
88 // connect to the specified socket.
89 return connectToSocket (host, port, timeout);
90 }
91
92 if (pipe.isNotEmpty ())
93 {
94 // else -- create and/or connect to a named pipe;
95 switch (options)
96 {
97 case ConnectOptions::createOrFail:
98 return createPipe (pipe, timeout, true);
99 case ConnectOptions::mustExist:
100 return connectToPipe (pipe, timeout);
101 case ConnectOptions::createIfNeeded:
102 return createPipe (pipe, timeout, false);
103 case ConnectOptions::noOptions:
104 default:
105 jassertfalse;
106 return false;
107 }
108 }
109 // invalid state -- no host *or* pipe defined.
110 jassertfalse;
111 return false;
112}

126void IpcClient::messageReceived (const juce::MemoryBlock& message)
127{
128 if (update & UpdateType::receive)
129 {
130 // create an rvalue copy of the message that can be moved from.
131 pushUpdate (juce::MemoryBlock { message });
132 clientProperties.rxCount++;
133 }

135void IpcClient::stateChanged (const void* encodedChange, size_t encodedSize)
136{
137 if ((update & UpdateType::send) && (clientProperties.connected))
138 {
139 if (updateData != SyncData { encodedChange, encodedSize })
140 {
141 sendMessage ({ encodedChange, encodedSize });
142 clientProperties.txCount++;
143 }
144 }

191IpcServer::IpcServer (Object& sync, IpcClient::UpdateType updateType, const juce::String& statePath, Object* state)
192: syncObject { sync }
193, update { updateType }
194, serverProperties { statePath, state }
195{
196 // the server properties will change its portNumber member to let us
197 // know that we should start or stop ourselves.
198 serverProperties.portNumber.onPropertyChange (
199 [this] (const juce::Identifier& /*id*/)
200 {
201 if (serverProperties.portNumber > 0)
202 startServer (serverProperties.portNumber, serverProperties.bindAddress);
203 else
204 stopServer ();
205 });
206}

216bool IpcServer::startServer (int portNumber, const juce::String& bindAddress)
217{
218 if (serverProperties.running)
219 {
220 serverProperties.status = IpcServerStatus::alreadyRunning;
221 return true;
222 }
223
224 if (beginWaitingForSocket (portNumber, bindAddress))
225 {
226 serverProperties.running = true;
227 serverProperties.status = IpcServerStatus::startedOkay;
228 return true;
229 }
230
231 serverProperties.running = false;
232 serverProperties.status = IpcServerStatus::errorStarting;
233 return false;
234}
235
236bool IpcServer::stopServer ()
237{
238 if (!serverProperties.running)
239 {
240 serverProperties.status = IpcServerStatus::alreadyStopped;
241 return true;
242 }
243
244 stop ();
245 // the stop() method returns void, so we have no way to verify that
246 // the server actually did stop!
247
248 serverProperties.running = false;
249 serverProperties.status = IpcServerStatus::stoppedOkay;
250 return true;
251}

253juce::InterprocessConnection* IpcServer::createConnectionObject ()
254{
255 // create a new IpcConnection object, and take over its ownership;
256 // pass back a non-owning pointer to it so the base server class can
257 // finish setting up the client connection.
258 auto client { std::make_unique<IpcClient> (syncObject, "", 0, 0, update, &serverProperties) };
259 juce::InterprocessConnection* connection { client.get () };
260 connections.push_back (std::move (client));
261 return connection;
262}
