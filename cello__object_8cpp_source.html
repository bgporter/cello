95bool Object::upsert (const Object* object, const juce::Identifier& key, bool deep)
96{
97 if (!object->hasattr (key))
98 return false;
99
100 const auto val { object->data[key] };
101
102 auto existingItem { data.getChildWithProperty (key, val) };
103 if (existingItem.isValid ())
104 {
105 // we found the match -- update in place.
106 if (deep)
107 existingItem.copyPropertiesAndChildrenFrom (*object, getUndoManager ());
108 else
109 existingItem.copyPropertiesFrom (*object, getUndoManager ());
110 return true;
111 }
112 // else, we need to add a copy to the end of our children.
113 data.appendChild (object->clone (deep), getUndoManager ());
114 return true;
115}

117void Object::upsertAll (const Object* parent, const juce::Identifier& key, bool deep)
118{
119 juce::ValueTree parentTree { *parent };

187void Object::insert (Object* object, int index)
188{
189 if (object == this)
190 {
191 // can't add an object to itself!
192 jassertfalse;
193 return;
194 }
195 // a value tree can only have 1 parent -- if the new object has a parent,
196 // remove it there first.
197 juce::ValueTree newChild { *object };
198 juce::ValueTree parent { newChild.getParent () };
199
200 if (parent.isValid ())
201 {
202 // we can get into a weird state if we try to mix operations on
203 // different undo managers.
204 jassert (getUndoManager () == object->getUndoManager ());
205 parent.removeChild (newChild, getUndoManager ());
206 }
207 data.addChild (*object, index, getUndoManager ());
208 // make sure that the new child is using this object's undo manager.
209 object->setUndoManager (getUndoManager ());
210}

276juce::ValueTree Object::load (juce::File file, FileFormat format)
277{
278 if (format == Object::FileFormat::xml)
279 {
280 const auto xmlText { file.loadFileAsString () };
281 return juce::ValueTree::fromXml (xmlText);
282 }
283
284 // one of the binary formats
285 juce::MemoryBlock mb;
286 if (!file.loadFileAsData (mb))
287 {
288 jassertfalse;
289 return {};
290 }
291 if (format == Object::FileFormat::binary)
292 return juce::ValueTree::readFromData (mb.getData (), mb.getSize ());
293 else if (format == Object::FileFormat::zipped)
294 return juce::ValueTree::readFromGZIPData (mb.getData (), mb.getSize ());
295
296 // unknown format
297 jassertfalse;
298 return {};
299}

301juce::Result Object::save (juce::File file, FileFormat format) const
302{
303 if (format == FileFormat::xml)
304 {
305 auto res { file.create () };
306 if (res.wasOk ())
307 {
308 if (file.replaceWithText (data.toXmlString ()))
309 return juce::Result::ok ();
310 res = juce::Result::fail ("Error writing to " + file.getFullPathName ());
311 }
312 return res;
313 }
314
315 juce::FileOutputStream fos { file };
316 if (!fos.openedOk ())
317 {
318 jassertfalse;
319 return juce::Result::fail ("Unable to open " + file.getFullPathName () +
320 " for writing");
321 }
322
323 if (format == FileFormat::binary)
324 {
325 data.writeToStream (fos);
326 return juce::Result::ok ();
327 }
328
329 else if (format == FileFormat::zipped)
330 {
331 juce::GZIPCompressorOutputStream zipper { fos };
332 data.writeToStream (zipper);
333 return juce::Result::ok ();
334 }
335
336 // unknown format
337 jassertfalse;
338 return juce::Result::fail ("Unknown file format");
339}

341Object::CreationType Object::wrap (const juce::String& type, juce::ValueTree tree)
342{
343 creationType = CreationType::wrapped;
344#if PATH_IMPL
345 Path path { type };
346 // DBG(tree.toXmlString());
347 data = path.findValueTree (tree, Path::SearchType::createAll, nullptr);
348 // DBG(data.toXmlString());
349 if (path.getSearchResult () == Path::SearchResult::created)
350 creationType = CreationType::initialized;
351#else
352 if (tree.isValid ())
353 {
354 // case 1: We're passed the tree we use as our store directly.
355 if (tree.getType () == type)
356 data = tree;
357 else
358 {
359 // case 2: look in the state tree for our data.
360 auto childTree = tree.getChildWithName (type);
361 if (childTree.isValid ())
362 {
363 data = childTree;
364 }
365 else
366 {
367 // case 3: the state tree doesn't have a tree for our data type
368 // yet. Create an empty tree of the correct type, mark ourselves
369 // as having been default-initialized.
370 data = juce::ValueTree (type);
371 creationType = CreationType::initialized;
372 tree.appendChild (data, getUndoManager ());
373 }
374 }
375 }
376 else
377 {
378 // case 4: There's no state, just create an empty tree of the correct
379 // type and mark ourselves as needing to be initialized.
380 data = juce::ValueTree (type);
381 creationType = CreationType::initialized;
382 }
383#endif
384
385 // register to receive callbacks when the tree changes.
386 data.addListener (this);
387 return creationType;
388}

390void Object::valueTreePropertyChanged (juce::ValueTree& treeWhosePropertyHasChanged,
391 const juce::Identifier& property)
392{
393 if (treeWhosePropertyHasChanged == data)
394 {
395 // first, try to find a callback for that exact property.
396 for (const auto& updater : propertyUpdaters)
397 {
398 if (updater.id == property)
399 {
400 if (updater.fn != nullptr)
401 updater.fn (property);
402 return;
403 }
404 }
405 // a cello extension: register a callback on the name of the tree's
406 // type, and you'll get a callback there for any property change that
407 // didn't have its own callback registered.
408 if (property != getType ())
409 valueTreePropertyChanged (data, getType ());
410 }
411}
