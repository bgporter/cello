67template <typename T> class Value : public ValueBase
68{
69public:
77 Value (Object& data, const juce::Identifier& id_, T initVal = {})
78 : ValueBase { id_ }
79 , object { data }
80 {
81 // if the object doesn't have this value yet, add it and set it
82 // to the initial value. This will happen as part of initializing a
83 // new Object, but may also happen if new values are added to an existing
84 // type.
85 if (!object.hasattr (id))
86 object.setattr<T> (id, initVal);
87 }

97 Value& operator= (const T& val)
98 {
99 set (val);
100 return *this;
101 }

110 void set (const T& val)
111 {
112 if (onSet != nullptr)
113 doSet (onSet (val));
114 else
115 doSet (val);
116 }

123 operator T () const
124 {
125 if (onGet != nullptr)
126 return onGet (doGet ());

145 class Cached
146 {
147 public:
148 Cached (Value<T>& val)
149 : value { val }
150 , cachedValue { static_cast<T> (value) }
151 {
152 // when the underlying value changes, cache it here so it can
153 // be used without needing to look it up, go through validation, etc.
154 value.onPropertyChange ([this] (juce::Identifier id)
155 { cachedValue = static_cast<T> (value); });
156 }
157
158 ~Cached () { value.onPropertyChange (nullptr); }
159
160 operator T () const { return cachedValue; }
161
162 private:
163 Value<T>& value;
164 T cachedValue;
165 };

171 Cached getCached () { return Cached (*this); }

180 using ValidatePropertyFn = std::function<T (const T&)>;
184 ValidatePropertyFn onSet;

209 void onPropertyChange (PropertyUpdateFn callback)

214private:
215 void doSet (const T& val)
216 {
217 juce::ValueTree tree { object };
218
219 // check if this call should change the current value.
220 if (notEqualTo (val))
221 {
222 // check if this value or our parent object have a listener to exclude
223 // from updates.
224 auto* excluded = (excludedListener != nullptr)
225 ? excludedListener
226 : object.getExcludedListener ();
227 const auto asVar { juce::VariantConverter<T>::toVar (val) };
228 if (excluded)
229 tree.setPropertyExcludingListener (excluded, id, asVar,
230 object.getUndoManager ());
231 else
232 tree.setProperty (id, asVar, object.getUndoManager ());
233 }
234 else
235 {
236 // check if we or our parent object want us to always send
237 // a property change callback for this value.
238 const auto forceUpdate = shouldForceUpdate () || object.shouldForceUpdate ();
239 if (forceUpdate)
240 tree.sendPropertyChangeMessage (id);
241 }
242 }
243
244 T doGet () const
245 {
246 juce::ValueTree tree { object };
247 return juce::VariantConverter<T>::fromVar (tree.getProperty (id));
248 }
249
257 bool notEqualTo (const T& newValue)
258 {
259 if constexpr (std::is_floating_point_v<T>)
260 return std::fabs (newValue - doGet ()) > epsilon;
261 else
262 return (newValue != doGet ());
263 }
264
265public:
268 static inline float epsilon { 0.001f };
269
270private:
272 Object& object;
273
275 juce::ValueTree::Listener* excludedListener { nullptr };
276};
