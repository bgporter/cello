146 void doSet (const T& val)
147 {
148 juce::ValueTree tree { object };
149
150 // check if this call should change the current value.
151 if (notEqualTo (val))
152 {
153 // check if this value or our parent object have a listener to exclude
154 // from updates.
155 auto* excluded = (excludedListener != nullptr)
156 ? excludedListener
157 : object.getExcludedListener ();
158 const auto asVar { juce::VariantConverter<T>::toVar (val) };
159 if (excluded)
160 tree.setPropertyExcludingListener (excluded, id, asVar,
161 object.getUndoManager ());
162 else
163 tree.setProperty (id, asVar, object.getUndoManager ());
164 }
165 else
166 {
167 // check if we or our parent object want us to always send
168 // a property change callback for this value.
169 const auto forceUpdate = shouldForceUpdate () || object.shouldForceUpdate ();
170 if (forceUpdate)
171 tree.sendPropertyChangeMessage (id);
172 }
173 }
174
175 T doGet () const
176 {
177 juce::ValueTree tree { object };
178 return juce::VariantConverter<T>::fromVar (tree.getProperty (id));
179 }

188 bool notEqualTo (T newValue)
189 {
190 if constexpr (std::is_floating_point_v<T>)
191 return std::fabs (newValue - doGet ()) > epsilon;
192 else
193 return (newValue != doGet ());
194 }
