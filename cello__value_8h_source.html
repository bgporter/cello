67template <typename T> class Value : public ValueBase
68{
69public:
77 Value (Object& data, const juce::Identifier& id, T initVal = {})
78 : ValueBase { id }
79 , object { data }
80 {
81 // if the object doesn't have this value yet, add it and set it
82 // to the initial value. This will happen as part of initializing a
83 // new Object, but may also happen if new values are added to an existing
84 // type.
85 if (!object.hasattr (id))
86 object.setattr (id, initVal);
87 }

160private:
161 void doSet (const T& val)
162 {
163 juce::ValueTree tree { object };
164
165 // check if this call should change the current value.
166 if (notEqualTo (val))
167 {
168 // check if this value or our parent object have a listener to exclude
169 // from updates.
170 auto* excluded = (excludedListener != nullptr)
171 ? excludedListener
172 : object.getExcludedListener ();
173 const auto asVar { juce::VariantConverter<T>::toVar (val) };
174 if (excluded)
175 tree.setPropertyExcludingListener (excluded, id, asVar,
176 object.getUndoManager ());
177 else
178 tree.setProperty (id, asVar, object.getUndoManager ());
179 }
180 else
181 {
182 // check if we or our parent object want us to always send
183 // a property change callback for this value.
184 const auto forceUpdate = shouldForceUpdate () || object.shouldForceUpdate ();
185 if (forceUpdate)
186 tree.sendPropertyChangeMessage (id);
187 }
188 }
189
190 T doGet () const
191 {
192 juce::ValueTree tree { object };
193 return juce::VariantConverter<T>::fromVar (tree.getProperty (id));
194 }
195
203 bool notEqualTo (T newValue)
204 {
205 if constexpr (std::is_floating_point_v<T>)
206 return std::fabs (newValue - doGet ()) > epsilon;
207 else
208 return (newValue != doGet ());
209 }
210
211public:
214 static inline float epsilon { 0.001f };
215
216private:
218 Object& object;
219
221 juce::ValueTree::Listener* excludedListener { nullptr };
222};
